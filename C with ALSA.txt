///write the C code to play beep on the selected card in int main to connect with ALSA

#include <stdio.h>
#include <math.h>
#include <alsa/asoundlib.h>

#define SAMPLE_RATE 48000
#define CHANNELS 2
#define FRAMES 256
#define DURATION_SEC 3
#define FREQ 440.0

int main(void)
{
    snd_pcm_t *handle;
    snd_pcm_hw_params_t *params;
    unsigned int rate = SAMPLE_RATE;
    int dir = 0;
    int rc;

    rc = snd_pcm_open(&handle, "hw:2,0", SND_PCM_STREAM_PLAYBACK, 0);
    if (rc < 0) {
        fprintf(stderr, "Unable to open audio device: %s\n", snd_strerror(rc));
        return 1;
    }

    snd_pcm_hw_params_malloc(&params);
    snd_pcm_hw_params_any(handle, params);
    snd_pcm_hw_params_set_access(handle, params, SND_PCM_ACCESS_RW_INTERLEAVED);
    snd_pcm_hw_params_set_format(handle, params, SND_PCM_FORMAT_S16_LE);
    snd_pcm_hw_params_set_channels(handle, params, CHANNELS);
    snd_pcm_hw_params_set_rate_near(handle, params, &rate, &dir);

    snd_pcm_uframes_t frames = FRAMES;
    snd_pcm_hw_params_set_period_size_near(handle, params, &frames, &dir);

    rc = snd_pcm_hw_params(handle, params);
    snd_pcm_hw_params_free(params);

    if (rc < 0) {
        fprintf(stderr, "Unable to set hardware parameters: %s\n", snd_strerror(rc));
        snd_pcm_close(handle);
        return 1;
    }

    rc = snd_pcm_prepare(handle);
    if (rc < 0) {
        fprintf(stderr, "Unable to prepare audio device: %s\n", snd_strerror(rc));
        snd_pcm_close(handle);
        return 1;
    }

    short buffer[FRAMES * CHANNELS];
    int loops = (SAMPLE_RATE * DURATION_SEC) / FRAMES;

    for (int i = 0; i < loops; i++) {
        for (int j = 0; j < FRAMES; j++) {
            double t = (double)(i * FRAMES + j) / SAMPLE_RATE;
            short sample = (short)(12000.0 * sin(2.0 * M_PI * FREQ * t));
            buffer[j * 2] = sample;
            buffer[j * 2 + 1] = sample;
        }

        rc = snd_pcm_writei(handle, buffer, FRAMES);
        if (rc == -EPIPE) {
            snd_pcm_prepare(handle);
        } else if (rc < 0) {
            fprintf(stderr, "Write error: %s\n", snd_strerror(rc));
            break;
        }
    }

    snd_pcm_drain(handle);
    snd_pcm_close(handle);

    printf("Done playing tone\n");
    return 0;
}
